#include<bits/stdc++.h>
using namespace std;
int main(){
    
    // write your code here
    int t;
    cin>> t;
    while(t--){
        int n,m;
        cin >> n >> m;
        vector<vector<long long >> mat (n, vector<long long >(m));
        
        for( int i = 0 ; i< n; i++){
            for ( int j =0 ; j < m ; j++){
                cin >> mat[i][j];
                
            }
        }
        
        int k ; 
        
        cin>> k ;
        int ql = k;
        vector<pair<int,int>> p;
        while(k--){
            int a; 
            int b;
            
            cin >>a >> b;
            p.push_back(make_pair(a,b));
            
        }
        
          vector<vector<long long >> dp (n, vector<long long >(m,-1));
        
          for( int i = 0 ; i< n; i++){
            for ( int j =0 ; j < m ; j++){
                if(i == 0){
                    dp[i][j] = 1;
                }else{
                    if(mat[i-1][j] < mat[i][j]){
                        dp[i][j] =   dp[i-1][j] ;
                    }else{
                        dp[i][j] = i+1;
                    }
                }
                
                
            }
        }
        // for( int i = 0 ; i< n; i++){

        // }
        bool flag = false;
        for(int i = 0; i< ql; i++){
        	for ( int j =0 ; j < m; j++){
            	if( dp[p[i].first-1][j] ==  dp[p[i].second-1][j]){
                    flag = true ;
                    break;
                }
              
                
            }
        
              if(flag)
            cout<< "Yes"<<endl;
            else
             cout<< "No"<<endl;
            flag =false;
        }
      
        
        
    }
    return 0;
}
