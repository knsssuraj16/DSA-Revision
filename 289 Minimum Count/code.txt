#include<bits/stdc++.h>
using namespace std;
int dp[1001];
int minCount(int n){
    if(n == 0){
        return 0;
    }
    if(n < 0){
        return INT_MAX;
    }
    
    if(dp[n] != -1){
        return dp[n];
    }
    int mn = INT_MAX;
    for(int i =1 ; i*i<=n;i++){
        mn =  min(minCount(n-(i*i)),mn);
    }
    
    dp [n] = mn+1; 
    return mn+1;
}
int main(){
    
    // write your code here
    
    int t;
    cin>>t;
    while(t--){
        int n;
        cin >> n;
        memset(dp,-1,sizeof dp);
        cout<<minCount(n)<<endl;
    }
    return 0;
}
