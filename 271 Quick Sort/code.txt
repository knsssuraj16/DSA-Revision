#include<bits/stdc++.h>
using namespace std;
int part(long long  arr[],int st,int en){
    long long pivot = arr[en];
    int i = st-1;
    for(int j=st;j<= en-1;j++){
        if(arr[j] < pivot){
            i++;
            swap(arr[i],arr[j]);
        }
    }
    i++;
    swap(arr[i],arr[en]);
    return i;
}
void qs(long long arr[],int st,int en){
    if(st>= en){
        return;
    }
    int pa = part(arr,st,en);
    qs(arr,st,pa-1);
    qs(arr,pa+1,en);
}
int main(){
    
   	int t;
    cin>>t;
    while(t--){
        int n ;
        cin>>n;
        long long arr[n];
        for(int i =0;i<n;i++){
            cin>>arr[i];
        }
        qs(arr,0,n);
        for(int i =0;i<n;i++){
            cout <<arr[i]<<" ";
        }
        cout<<endl;
        
        
    }
    return 0;
}
