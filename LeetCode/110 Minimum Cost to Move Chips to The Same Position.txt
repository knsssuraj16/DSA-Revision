class Solution {
public:
    int minCostToMoveChips(vector<int>& position) {
        int ev =  0,odd =0 ;
        for(int i =0 ;i < position.size(); i++){
            if( (position[i] & 1) == 0){
                ev +=1 ;
            }else{
                odd +=1;
            }
        }
        
        return min(ev,odd);
    }
};
