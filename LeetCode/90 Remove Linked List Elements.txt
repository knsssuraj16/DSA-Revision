    ListNode* removeElements(ListNode* head, int val) {
        
        ListNode* dummy= new ListNode(-1,head);
        
        ListNode * temp= dummy;
        
        while(temp != NULL && temp->next != NULL){
            if(temp->next->val == val){
                temp->next = temp->next->next;
            }else
             temp = temp->next;
        }

        
        return dummy->next;
