using namespace std;

int kadane( vector<int>  &narr,int m){
    int res =narr[0] ;
    int finalres =INT_MIN;
    for(int i =1 ; i< m; i++){
    	if(res<0){
            res = narr[i];
        }
        else{
            res += narr[i];
        }
        
        finalres = max(finalres,res);
    }
    
    return finalres;
}

int solve(vector<vector<int>> &arr,int n,int m){
    int ans = INT_MIN;
    
    
    
    for(int k = 0;k< n; k++){
        
        vector<int> narr(m,0);
        
        for(int i =k;i< n;i++){
            for(int j =0; j< m; j++){
                
                narr[j] += arr[i][j];
                
            }
           ans = max(ans,kadane(narr,m));
        }
    }
    
    return ans;
}
