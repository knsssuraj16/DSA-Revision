#include<bits/stdc++.h>
using namespace std;
int dp[100005];
int solve(string &s,int n){
     
  
    if(n == 0){
        return 1;
    }
    if(n == 1){
        return 1;
    }
 
    if(dp[n] != -1){
        return dp[n];
    }
    
   if(s[n-1] == '0'){
        dp[n]=0;
        return 0;
    }
   
    long long res = (long long)solve(s ,n-1);
   
    long long res2 = 0;
    int n0= s[n-1]- '0' ;
    int n1= s[n-2] -'0';
    if(((n1 *10) +n0) <=26 ){
        res2 =(long long)solve(s ,n-2);
    }
    // cout<<"res: "<<res;
    // cout<<"res1: "<<res2;
    dp[n] = (int) (( res + res2) % 1000000007);
   return dp[n];
    
}

        // int *arr = new int[s.length()+1]();
          memset(dp,-1,sizeof dp);
       
        cin>>s;

        cout<<solve(s,s.length())<<endl;

}
