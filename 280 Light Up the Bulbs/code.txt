#include<bits/stdc++.h>
using namespace std;
int main(){
    
    // write your code here
    
    int n;
    cin>>n;
    long long r,t ;
    cin>>r>>t;
    string s;
    cin>>s;
    
    int ze=0;
    if(s[0] == '0'){
        ze++;
    }
    for(int i =0; i<s.length()-1;i++){
        if(s[i] =='1' && s[i+1] =='0'){
            ze++;
        }
    }
    
    long long res=0;
    if(ze == 1){
        res= t;
    }else if(ze>1){
        
    
        if(r<t){
         res=  ( ( ze -1 )* r) + t;
        }else{
            res= ze* t ;
        }
    }
    cout<<res<<endl;
    return 0;
}
