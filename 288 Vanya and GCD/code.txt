#include<bits/stdc++.h>
using namespace std;

#define m 1000000007
int solve (vector<int> & arr,int n ){
    
    vector<vector<long long >> dp(n+1,vector<long long>(101,0));
    
    
    dp[0][arr[0]] = 1;
    
    long long sum=dp[0][1];
    for(int i =1; i< n; i++){
        for(int j =0 ; j< i; j++){
            if(arr[i] > arr[j]){
                for(int k = 1; k< 101; k++){
                   int ng =  __gcd(arr[i],k);
                    
                    dp[i][ng] = (dp[i][ng] + dp[j][k]);
                }
            }
        }
        
        dp[i][arr[i]]++;
        sum = (sum + dp[i][1]);
    }
    
    // for(int i =0; i<n;i++){
    //     for(int j=0 ;j<4; j++){
    //         cout<< dp[i][j] <<" ";
    //     }
    //     cout<<endl;
    // }
    
    return sum % m ;
}
