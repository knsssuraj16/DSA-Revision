#include<bits/stdc++.h>
using namespace std;

void mer(int arr[],int st,int mid,int en){
    int i = st;
    int j= mid+1;
    int temp [en-st+1];
    int k=0;
    while(i<=mid && j <=en){
        if(arr[i]>arr[j]){
            temp [k++]= arr[j++];
        }else{
             temp [k++]= arr[i++];
        }
    }
    
    while(i<=mid){
         temp [k++]= arr[i++];
    }  
    
    while(j<=en){
         temp [k++]= arr[j++];
    }
    
    k=0;
    for(;st<=en;st++)
    {
        arr[st]= temp[k++];
    }
}

void mSort (int arr[],int st,int en){
    if(st>=en){
        return;
    }
    int mid = (st+en)/2;
    mSort(arr,st,mid);
    mSort(arr,mid+1,en);
    
    mer(arr,st,mid,en);
}
int main(){
    
    int t;
    cin>>t;
    while(t--){
        int n;
        cin>>n;
        int arr[n];
        for(int i=0;i<n;i++){
            cin>>arr[i];
            
        }
        mSort(arr,0,n-1);
        for(int i=0;i<n;i++){
            cout<<arr[i]<<" ";
            
            
        }
        cout<<endl;
    }
    return 0;
}
